#include <iostream>
#include <iomanip>
#include <vector>

using namespace std;

struct CW
{
	char c;  // 字符
	int  w;  // 权重
};
struct HuffNode
{
	char c;  // 字符
	int  w;  // 权重
	int lchild,rchild,parent; 
};

struct Node
{
	int w;  // 权重
	int i;  // 下标
	Node(int w,int i):w(w),i(i){}
};
bool operator>(const Node &n1,const Node &n2)
{
	return n1.w>n2.w;
}

// 编码个数 = n
// 时间复杂度：O(n*n) -> O(n*log2(n))
// 空间复杂度：O(1)   -> O(n)
void CreateHT(CW cw[],int n, HuffNode HT[])
{
	priority_queue<Node,vector<Node>, greater<Node> >  Q; // 小根堆
	int i=0;
	for(; i<n; i++)
	{
		HT[i].c=cw[i].c,
		HT[i].w=cw[i].w,
		HT[i].lchild=HT[i].rchild=HT[i].parent=-1;
		Q.push( Node(HT[i].w, i) );  // 进队列
	}
	for(; i<2*n-1; i++)//哈夫曼树必然只有两种节点，一种是叶子结点，一种是有两个分支的节点，因为二叉树性质n0=n2+1
	{
		HT[i].c = ' ';
		HT[i].parent = -1;
		// 优先队列中出队列，找权值最小的根结点的下标min1,min2
		Node min1 = Q.top();Q.pop();
		Node min2 = Q.top();Q.pop();		
		HT[min1.i].parent=i; HT[min2.i].parent=i;  // 认爹
		HT[i].lchild=min1.i, HT[i].rchild=min2.i;  // 爹认左右孩子
		HT[i].w = HT[min1.i].w + HT[min2.i].w;
		Q.push( Node(HT[i].w, i) );
	}
}

// 编码个数 = n
// 时间复杂度：O(n*n) -> O(n*log2(n))
// 空间复杂度：O(1)   -> O(n)
void CreateHT(CW cw[],int n, HuffNode HT[])
{
	priority_queue<HuffNode,vector<HuffNode>,greater<HuffNode> >Q;
	int i=0;
	for(;i<n;i++){
		HT[i].c=cw[i].c;
		HT[i].w=cw[i].w;
		HT[i].i=i;
		HT[i].l=HT[i].r=HT[i].parent=-1;
		Q.push(HT[i]);
	}
	int tt=n;
	while(!Q.empty()){
		HuffNode min1=Q.top();Q.pop();
		HuffNode min2=Q.top();Q.pop();
		HuffNode newnode;
		newnode.w=min1.w+min2.w;
		newnode.c='o';//占位字符
		newnode.i=tt++;
		HT[min1.i].parent=HT[min2.i].parent=newnode.i;
		newnode.l=min1.i;newnode.r=min2.i;
		HT[tt-1]=newnode;
		Q.push(newnode);
	}
}

class Heap
{
	int a[N],n;
public:
	Heap():n(0){ }
	//已知a[0]...a[n-1]是小根堆，插入了a[n]
	void push(int x)
	{
		a[n++]=x;
		int p=n-1, parent= (p-1)/2;
		while(p>0)
		{
			if(a[parent]<=a[p])  return;
			swap(a[parent], a[p]);
			p=parent, parent= (p-1)/2;
		}
	}
	void pop()
	{
		a[0]=a[n-1], n--;
		int p=0;
		while( 2*p+1<n )//判左是否越界
		{
			int child=2*p+1; // child是左右孩子中最小值的下标，因为从0开始，所以左孩子是2*p+1,右孩子是2*p+2.
			if(child+1<n && a[child+1]<a[child])//判右是否越界
				child++;
			if(a[p]<=a[child])  break;
			swap(a[p], a[child]);
			p=child;
		}
	}
	int top(){ return a[0]; }
	bool empty(){ return n==0; }
}

int partition( int left, int right,vector<int>& num) {  
    int pivot = num[right];  // 选择最右边的元素作为pivot  
    int i = left - 1;  // i用来追踪大于pivot的元素的最后一个索引位置  
	for (int j = left; j < right; ++j) {  
		if (num[i] > pivot) {  // 如果当前元素大于pivot  
			++i;  // 移动到下一个索引  
			swap(num[i], num[j]);  // 交换大于pivot的元素和当前元素
		}  
	}  

    // 最后交换pivot和第一个小于pivot的元素  
    swap(num[i+1], num[right]);  
    return i + 1;  // 返回pivot的索引  
}  
  
int findk(vector<int>&num,int l,int r,int k){  
    int mid=partition(l,r,num);  
    if(mid==k-1) return num[mid];//如果相等就返回  
    else if(mid<k-1){  
        return findk(num,mid+1,r);  
    }     
    else{  
        return findk(num,l,mid-1);  
    }  
}

// 在sd[]找最小值的下标imin，要求set[imin]=false
int FindiMin(int sd[],bool set[], int n)
{
	int imin=0;
	while(set[imin]==true) imin++;
	for(int i=imin+1; i<n; i++)
	{
		if(set[i]==true)  continue;
		if(sd[i]<sd[imin]) imin=i;
	}
	return imin;
}

void Dijkstra(int M[N][N],int n, int sv,int sd[],int prev[])
{
	bool set[N];  
	for(int i=0;i<n; i++)
		sd[i]=M[sv][i], set[i]=false, prev[i]=-1;
	set[sv]=true;
	// 找最近的顶点，调整sd[],set[]
	for(i=0; i<n-1; i++)
	{
		int w=FindiMin(sd,set, n);	set[w]=true; // sv->w的最短距离已经确定
		// 借助w,以w为中转站，优化sd[]
		for(int j=0; j<n; j++)
		{
			if(set[j]==true)  continue; // 忽略已经确定最短距离的顶点
			if(sd[w]+M[w][j] < sd[j])
				sd[j] = sd[w]+M[w][j],  // 贪心法：找到了更近的路
				prev[j] = w;  // 记住sv->j的路上最后一个中间顶点
		}
	}
}

void Dijkstra(int M[N][N],int n, int sv,int sd[],int prev[])
{
	bool set[N]={false};
	for(int i=0;i<n;i++){
		sd[i]=M[sv][i];prev[i]=-1;
	}
	for(int i=0;i<n;i++){
		int imin=FindMin(sd,set,n);
		//更新set
		set[imin]=true;
		//更新sd
		for(int j=0;j<n;j++){
			if(set[j]==true)continue;
			if(sd[j]>sd[imin]+M[imin][j])
				sd[j]=sd[imin]+M[imin][j];
				prev[j]=imin;
		}
	}
	
}

// C = A + B, A >= 0, B >= 0
vector<int> add(vector<int> &A, vector<int> &B)
{
    if (A.size() < B.size()) return add(B, A);

    vector<int> C;
    int t = 0;
    for (int i = 0; i < A.size(); i ++ )
    {
        t += A[i];
        if (i < B.size()) t += B[i];
        C.push_back(t % 10);
        t /= 10;
    }

    if (t) C.push_back(t);
    return C;
}

// C = A - B, 满足A >= B, A >= 0, B >= 0
vector<int> sub(vector<int> &A, vector<int> &B)
{
    vector<int> C;
    for (int i = 0, t = 0; i < A.size(); i ++ )
    {
        t = A[i] - t;
        if (i < B.size()) t -= B[i];
        C.push_back((t + 10) % 10);
        if (t < 0) t = 1;
        else t = 0;
    }
    while (C.size() > 1 && C.back() == 0) C.pop_back();
    return C;
}

vector<int> mul(vector<int> &A, int b)
{
    vector<int> C;

    int t = 0;
    for (int i = 0; i < A.size() || t; i ++ )
    {
        if (i < A.size()) t += A[i] * b;
        C.push_back(t % 10);
        t /= 10;
    }
    while (C.size() > 1 && C.back() == 0) C.pop_back();
    return C;
}

// A / b = C ... r, A >= 0, b > 0
vector<int> div(vector<int> &A, int b, int &r)
{
    vector<int> C;
    r = 0;
    for (int i = A.size() - 1; i >= 0; i -- )
    {
        r = r * 10 + A[i];
        C.push_back(r / b);
        r %= b;
    }
    reverse(C.begin(), C.end());
    while (C.size() > 1 && C.back() == 0) C.pop_back();
    return C;
}

vector<int> spiralOrder(vector<vector<int>>& matrix) {
        int row=matrix.size();
        int col=matrix[0].size();
        vector<vector<bool>> flag(row,vector<bool>(col));
        pair<int,int> dirct[4]={{0,1},{1,0},{0,-1},{-1,0}};//右下左上
        bool stop=false; int i=0,j=0;
        vector<int>res;
        
        res.push_back(matrix[i][j]);
        flag[0][0]=true;

        while(!stop){
            stop = true;  // 假设所有的位置都已经遍历结束
            for(int t = 0; t < 4; t++)
            {
                while (true)  // 尝试按照当前的方向前进
                {
                    int new_i = i + dirct[t].first;
                    int new_j = j + dirct[t].second;
                    // 如果新的位置越界或者已经被访问过，就换一个新的方向
                    if (new_i < 0 || new_i >= row || new_j < 0 || new_j >= col || flag[new_i][new_j] == true) {
                        break;
                    }
                    // 否则，前进到新的位置，并标记新的位置已被访问
                    i = new_i;
                    j = new_j;

                    res.push_back(matrix[i][j]);
                    cout << matrix[i][j] << ' ';
                    flag[i][j] = true;
                    stop = false;  // 只要有新的位置被访问，就表示还没有遍历结束
                }
            }
        }
        return res;
    }

int Solve(int vs[],int n, int Vol)
{
	vector<int> dp(Vol+1,0);
	vector<int> prev(Vol+1,-1);
	for(int v=1; v<=Vol; v++)
	{
		for(int i=0; i<n; i++)  // 尝试放入vs[i]
		{
			if(vs[i]>v)    continue;  // 放不进去
			if(vs[i] + dp[v-vs[i]]>dp[v])
				dp[v] = vs[i] + dp[v-vs[i]], prev[v]=v-vs[i];
		}
	}
	//Output(prev);
	Output(prev, Vol); // 输出方案
	return dp.back();
}

/*

	    空   0  0  0  0  0  0  0
	X1[]={1, 0  0  0  0  1  1  1
		  2, 0  1  1  1  1  2  2
		  3, 0  1  1  2  2  2  2
		  2, 0  1  1  2  2  3  3
		  4, 0  1  2  2  2  3  3  
		  1, 0  1  2  2  3  3  4
		  2: 0  1  2  2  3  4  4
*/
int FindLCS(int X1[],int n1,int X2[],int n2)
{
	vector<vector<int> > M(n1+1,vector<int>(n2+1,0) );
	for(int i=1; i<=n1; i++)
		for(int j=1; j<=n2; j++)
			if(X1[i-1] == X2[j-1])
				M[i][j] = M[i-1][j-1]+1;
			else
				if(M[i-1][j]>M[i][j-1])
					M[i][j] = M[i-1][j];
				else
					M[i][j] = M[i][j-1];
	return M[n1][n2];
}

	int X1[]={1,2,3,2,4,1,2};  // 2 3 1 2 
	int X2[]={2,4,3,1,2,1};    // 2 3 1 2 
	int L=FindLCS(X1,7,X2,6);

class Graph
{
	int vn; // 顶点个数
	vector<vector<int> > M;  // 代价矩阵
public:
	Graph(char fname[])
	{
		ifstream fin(fname);
		fin>>vn;
		M.resize(vn, vector<int>(vn, 0));
		for(int i=0; i<vn; i++)
			for(int j=0; j<vn; j++)
				fin>>M[i][j];
	}
	bool isOK(vector<int> &cs, int c)
	{
		int n=cs.size();  // 已经着色的顶点数
		// 当前给区域n 赋色c
		for(int i=0; i<n; i++)
		{
			if(M[n][i]==0)  continue; // 无边相连
			if(cs[i]==c)  return  false; // 颜色冲突
		}
		return true;
	}
	void dfs(int ith, vector<int> &cs)
	{
		if(ith==vn)
			Output(cs);
		else
		{
			for(int c=1; c<=4; c++)
			{
				if(isOK(cs,c)==false)  continue;   // 颜色c不合法，则舍弃
				cs.push_back(c);  // 给ith区域赋色c
				dfs(ith+1, cs);
				cs.pop_back();    // 撤销ith区的颜色
			}
		}
	}
	friend ostream & operator<<(ostream &out, Graph &g)
	{
		for(int i=0; i<g.vn; i++)
		{
			for(int j=0; j<g.vn; j++)
				out<<setw(3)<<g.M[i][j]<<"  ";
			out<<endl;
		}
		return out;
	}
};
int main()
{
	Graph G("g1.txt");
	vector<int> cs;  // 颜色表：cs[i]表示区域i的颜色
	G.dfs(0, cs);
	return 0;	
}

vector<bool> visited;

void permutation(int n, vector<int>& tmp) {
	if (tmp.size() == n) {
		output(tmp);
		return;
	}
	for (int i = 1; i <= n; i++) {
		if (!visited[i]) {
			visited[i] = true;
			tmp.push_back(i);
			permutation(n, tmp);
			tmp.pop_back();
			visited[i] = false;
		}
	}
}

struct Edge
{
	int sv,ev;

	Edge(int sv=0,int ev=0,int cost=0): sv(sv),ev(ev),cost(cost){}

		return in>>e.sv>>e.ev>>e.cost;
	}

	{
		return out<<e.sv<<"<->"<<e.ev<<": "<<e.cost;
	}

bool operator<(const Edge &e1,const Edge &e2)
{
	return e1.cost<e2.cost;
}
void Output(vector<Edge> &es)
{
	for(int i=0; i<es.size(); i++)
		cout<<es[i]<<endl;
}

class Graph
{
	int vn; // 顶点个数
	vector<Edge> es;  // 边集
public:
	Graph(char fname[])
	{
		ifstream fin(fname);
		int en;   // 边数
		fin>>vn>>en;
		es.resize(en);
		for(int i=0; i<en; i++)
			fin>>es[i];
		sort(es.begin(), es.end());
	}
	friend ostream & operator<<(ostream &out, Graph &g)
	{
		for(int i=0; i<g.es.size(); i++)
			out<<g.es[i]<<endl;
		return out;
	}
	// 返回连通分量的编号：根结点的下标
	int FindParent(vector<int> &parent, int v)
	{
		while(parent[v]!=-1)
			v=parent[v];
		return v;
	}
	void Kruskal(vector<Edge> &mst)
	{
		// parent[i]: k是顶点i所属的连通分量的编号
		vector<int>  parent(vn, -1);
		for(int i=0; i<es.size(); i++)
		{
			// es[i].sv   es[i].ev 是否属于同一个连通分量
			int sv = FindParent(parent,es[i].sv);
			int ev = FindParent(parent,es[i].ev);
			if(sv==ev)   continue;
			mst.push_back(es[i]);
			parent[sv]=ev;   // 建立sv和ev之间的连通关系
		}
	}
};

	Edge(int ev=0,int cost=0): ev(ev),cost(cost){}
	friend istream & operator>>(istream &in, Edge &e)
	{
		return in>>e.ev>>e.cost;
	}
};
bool operator<(const Edge &e1,const Edge &e2)
{
	return e1.cost>e2.cost;
}

// 2->3
int GetCost(multimap<int,Edge> &G, int i,int j)
{
	multimap<int,Edge>::iterator it= G.find(i);
	if(it==G.end())  return Inf;
	while(it!=G.end() && it->first==i)
	{
		if(it->second.ev==j) 
			return it->second.cost;
		it++;	
	}
	return Inf;
}

void Dijkstra(multimap<int,Edge> &G,int n, int sv,int sd[],int prev[])
{
	priority_queue<Edge> Q, nextQ;    // 大根堆,得出小根堆的效果
	bool set[N];  
	for(int i=0; i<n; i++)
	{
		sd[i]=GetCost(G,sv,i), set[i]=false, prev[i]=-1;
		if(i==sv)   continue;   // 忽略sd->sd:0
		Q.push( Edge(i, sd[i]) );
	}
	set[sv]=true;   
	// 找最近的顶点，调整sd[],set[]
	for(i=0; i<n-1; i++)
	{
		Edge e =Q.top();  Q.pop();  // 优先队列取最小值O(log2(n))
		int w=e.ev;
		set[w]=true; // sv->w的最短距离已经确定
		// 借助w,以w为中转站，优化sd[]
		while( !Q.empty() )
		{
			Edge e1 =Q.top();  Q.pop();  
			int j=e1.ev;   // j是尚未确定最短距离的顶点
			int w2j=GetCost(G,w, j);
			if(sd[w]+ w2j < sd[j])
				sd[j] = sd[w]+w2j,  // 贪心法：找到了更近的路
				prev[j] = w;  // 记住sv->j的路上最后一个中间顶点
			nextQ.push( Edge(j, sd[j]) );
		}
		Q = nextQ;
	}
}
// 输出sv->v的所有中间顶点
void GetPath(int prev[], int v)
{
	while(v!=-1)
	{
		cout<<prev[v]<<" ";
		v=prev[v];   // p=p->next
	}
}
void Output(multimap<int,Edge> &G)
{
	for(multimap<int,Edge>::iterator it=G.begin(); it!=G.end(); it++)
		cout<<it->first<<"->"<<it->second.ev<<": "<<it->second.cost<<endl;
}

{
	multimap<int,Edge> G;   // 巨大的稀疏图
	freopen("G1.txt", "r", stdin);
	int vn;  cin>>vn;   // 顶点数

	int prev[N];   // prev[i]：sv->i之前的最后顶点的下标

	}

}
